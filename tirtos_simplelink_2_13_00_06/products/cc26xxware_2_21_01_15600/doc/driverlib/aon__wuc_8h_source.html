227 __STATIC_INLINE void
228 AONWUCMcuPowerDownConfig(uint32_t ui32ClkSrc)
229 {
230  uint32_t ui32Reg;
231 
232  //
233  // Check the arguments.
234  //
235  ASSERT((ui32ClkSrc == AONWUC_NO_CLOCK) ||
236  (ui32ClkSrc == AONWUC_CLOCK_SRC_LF));
237 
238  //
239  // Set the clock source for the MCU domain when in power down.
240  //
241  ui32Reg = HWREG(AON_WUC_BASE + AON_WUC_O_MCUCLK);
242  ui32Reg &= ~AON_WUC_MCUCLK_PWR_DWN_SRC_M;
243  HWREG(AON_WUC_BASE + AON_WUC_O_MCUCLK) = ui32Reg |
244  (ui32ClkSrc <<
245  AON_WUC_MCUCLK_PWR_DWN_SRC_S);
246 }

348 __STATIC_INLINE void
349 AONWUCMcuSRamConfig(uint32_t ui32Retention)
350 {
351  uint32_t ui32Reg;
352 
353  //
354  // Check the arguments.
355  //
356  ASSERT(ui32Retention & MCU_RAM_BLOCK_RETENTION);
357  ASSERT(!(ui32Retention & ~MCU_RAM_BLOCK_RETENTION));
358 
359  //
360  // Configure the retention.
361  //
362  ui32Reg = HWREG(AON_WUC_BASE + AON_WUC_O_MCUCFG) & ~MCU_RAM_BLOCK_RETENTION;
363  ui32Reg |= ui32Retention;
364  HWREG(AON_WUC_BASE + AON_WUC_O_MCUCFG) = ui32Reg;
365 }
