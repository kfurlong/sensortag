356 PRCMMcuUldoConfigure(uint32_t ui32Enable)
357 {
358  uint32_t ui32Val;
359 
360  //

362  //
363  ui32Val = HWREG(PRCM_BASE + PRCM_O_VDCTL);
364  if(ui32Enable)
365  {

367  }

372  HWREG(PRCM_BASE + PRCM_O_VDCTL) = ui32Val;

616 __STATIC_INLINE void
617 PRCMDomainEnable(uint32_t ui32Domains)
618 {
619  //
620  // Check the arguments.
621  //
622  ASSERT((ui32Domains & PRCM_DOMAIN_RFCORE) ||
623  (ui32Domains & PRCM_DOMAIN_VIMS));
624 
625  //
626  // Enable the clock domain(s).
627  //
628  if(ui32Domains & PRCM_DOMAIN_RFCORE)
629  {
630  HWREG(PRCM_BASE + PRCM_O_RFCCLKG) = PRCM_RFCCLKG_CLK_EN;
631  }
632  if(ui32Domains & PRCM_DOMAIN_VIMS)
633  {
634  HWREG(PRCM_BASE + PRCM_O_VIMSCLKG) = PRCM_VIMSCLKG_CLK_EN_M;
635  }
636 }
