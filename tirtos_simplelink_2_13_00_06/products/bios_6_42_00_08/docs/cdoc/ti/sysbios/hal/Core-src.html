    36    /*!
    37     *  ======== Core ========
    38     *  "Global Proxy" for target specific Core implementations
    39     */
    40    module Core inherits ti.sysbios.interfaces.ICore
    41    {
    42    
    43        @Macro
    44        override UInt hwiDisable();
    45    
    46        @Macro
    47        override UInt hwiEnable();
    48    
    49        @Macro
    50        override Void hwiRestore(UInt key);
    51    
    52        /*!
    53         *  ======== getId ========
    54         *  return the current core id
    55         */
    56        @Macro
    57        override UInt getId();
    58    
    59        /*!
    60         *  @_nodoc
    61         *  ======== interruptCore ========
    62         *  Cause an interrupt on a particular core.
    63         */
    64        @Macro
    65        override Void interruptCore(UInt coreId);
    66    
    67        /*!
    68         *  ======== lock ========
    69         *  acquire Inter-core lock.
    70         */
    71        @Macro
    72        override IArg lock();
    73    
    74        /*!
    75         *  ======== unlock ========
    76         *  release Inter-core lock.
    77         */
    78        @Macro
    79        override Void unlock();
    80    
    81    internal:   /* not for client use */
    82    
    83        /*! target/device-specific ICore implementation. */
    84        proxy CoreProxy inherits ti.sysbios.interfaces.ICore;
    85    }
