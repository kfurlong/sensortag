383 __STATIC_INLINE void
384 SysCtrlSystemReset( void )
385 {
386  // Disable CPU interrupts
387  CPUcpsid();
388  // Write reset register
389  HWREGBITW( AON_SYSCTL_BASE + AON_SYSCTL_O_RESETCTL, AON_SYSCTL_RESETCTL_SYSRESET_BITN ) = 1;
390  // Finally, wait until the above write propagates
391  while ( 1 ) {
392  // Do nothing, just wait for the reset (and never return from here)
393  }
394 }
